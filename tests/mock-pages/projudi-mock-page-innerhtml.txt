            function mostraOculta(
              idParteProcesso,
              idTabela,
              linhaCount,
              tipoInfo
            ) {
              var titulo;
              var conteudoHTML;
              span = document.getElementById(
                "span" + tipoInfo + idParteProcesso

                tabela.deleteRow(trEndAdv.rowIndex);
              }
